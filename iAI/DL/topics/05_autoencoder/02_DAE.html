def encoder(x, weights, biases):
    encoder = tf.add(tf.matmul(x, weights['encoder']), biases['encoder'])
    encoder = tf.nn.relu(encoder)

    latent = tf.add(tf.matmul(encoder, weights['latent']), biases['latent'])

    return latent

def decoder(latent, weights, biases):
    decoder = tf.add(tf.matmul(latent, weights['decoder']), biases['decoder'])
    decoder = tf.nn.relu(decoder)
    
    reconst = tf.add(tf.matmul(decoder, weights['reconst']), biases['reconst'])    
    # to improve reconstruction image performance, use a nonlinear activation with sigmoid
    reconst = tf.nn.sigmoid(reconst)
    
    return reconst
In [6]:
LR = 0.0001

latent = encoder(x, weights, biases)
reconst = decoder(latent, weights, biases)
loss = tf.square(tf.subtract(x, reconst))
loss = tf.reduce_mean(loss)

optm = tf.train.AdamOptimizer(LR).minimize(loss)

3.5. Build a Model¶

Encoder

  • Simple ANN (MLP) model
  • Use RELU for a nonlinear activation function
  • latent is not applied with a nonlinear activation function

Decoder

  • Simple ANN (MLP) model
  • Use RELU for a nonlinear activation function
  • reconst is not applied with a nonlinear activation function



In [16]:
def encoder(x, weights, biases):
    encoder = tf.add(tf.matmul(x, weights['encoder']), biases['encoder'])
    encoder = tf.nn.relu(encoder)

    latent = tf.add(tf.matmul(encoder, weights['latent']), biases['latent'])

    return latent
In [17]:
def decoder(latent, weights, biases):
    decoder = tf.add(tf.matmul(latent, weights['decoder']), biases['decoder'])
    decoder = tf.nn.relu(decoder)
    
    reconst = tf.add(tf.matmul(decoder, weights['reconst']), biases['reconst'])    
    # to improve reconstruction image performance, use a nonlinear activation with sigmoid
    reconst = tf.nn.sigmoid(reconst)
    
    return reconst

3.6. Define Loss, and Optimizer¶

Loss

  • Squared loss
$$ \frac{1}{m}\sum_{i=1}^{m} (t_{i} - y_{i})^2 $$

Optimizer

  • AdamOptimizer: the most popular optimizer
