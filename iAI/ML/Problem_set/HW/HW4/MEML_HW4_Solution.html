n_iter = 100000
n_stay = 0
n_switch = 0
doors = ['car', 'goat', 'goat']

print('Start simulation ...')
for i in range(n_iter):
    # Initializing game
    shuffle(doors)
    door_choice = choice([0, 1, 2])
    door_car = doors.index('car')
    
    # Making decision. Each decision would be selected with 1/2 probability
    # So we have to count it as twice.
    decision = choice(['stay', 'switch'])
    if decision == 'stay':
        if door_choice == door_car:
            n_stay += 2
    elif decision == 'switch':
        if door_choice != door_car:
            n_switch += 2

print('Staying: {}'.format(n_stay/n_iter))
print('Switching: {}'.format(n_switch/n_iter))
