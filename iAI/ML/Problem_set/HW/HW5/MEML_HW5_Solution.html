In [2]:
x_reg = x_reg.reshape(-1, 1)
y_reg = y_reg.reshape(-1, 1)
ones_col = np.ones((x_reg.size, 1))
phi = np.hstack([x_reg, ones_col])

w = cvx.Variable(2)
objective = cvx.Minimize(cvx.sum_squares(phi * w - y_reg))
prob = cvx.Problem(objective)
prob.solve()
w = w.value

plt.figure(figsize = (10, 8))
plt.plot(x_reg, y_reg, 'k.', label = 'data')

x0 = np.asmatrix(x0)
x1 = np.asmatrix(x1)

mu0 = np.mean(x0, axis = 0)
mu1 = np.mean(x1, axis = 0)
S0 = 1 / (n0 - 1) * (x0 - mu0).T * (x0 - mu0)
S1 = 1 / (n1 - 1) * (x1 - mu1).T * (x1 - mu1)

w = (n0 * S0 + n1 * S1).I * (mu0 - mu1).T
h = w[1] / w[0]
xp = np.arange(-2, 2, 0.1)
yp = (h * xp).reshape(-1, 1)

plt.figure(figsize = (10, 8))
plt.plot(x0[:,0],x0[:,1],'r.')
plt.plot(x1[:,0],x1[:,1],'b.')
plt.plot(xp, yp, 'b', linewidth = 4.0, label = 'FDA - projection line')
plt.legend()
plt.axis('equal')
plt.show()
